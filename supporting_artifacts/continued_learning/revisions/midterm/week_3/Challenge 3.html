Restructure Data

hip_hop <- hip_hop |> 
  mutate(ethnic_w = ifelse(ethnic=='white','white','non-white')
  ) |> 
  distinct(subj,.keep_all = TRUE) |> 
  subset(select = c(subj, sex, age, ethnic, intl, vocal, classical, folk,
                    rock, country, pop, alternative, hiphop, unclassifiable,
                    ethnic_w))
